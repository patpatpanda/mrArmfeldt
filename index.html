2. Hitta grannar vars summa är 9


int[] heltal = { 2, 7, 11, 15 };
int target = 9;

for (int i = 0; i < heltal.Length - 1; i++)
{
    if (heltal[i] + heltal[i + 1] == target)
    {
        Console.WriteLine($"Index {i} och {i + 1} ger summan {target}");
    }
}

3. Hitta alla par vars summa är 9 (Two Sum enkel version)


int[] heltal = { 2, 7, 11, 15 };
int target = 9;

for (int i = 0; i < heltal.Length; i++)
{
    for (int j = i + 1; j < heltal.Length; j++)
    {
        if (heltal[i] + heltal[j] == target)
        {
            Console.WriteLine($"Index {i} och {j} ger summan {target}");
        }
    }
}

4. Vänd ett ord baklänges med for-loop


string ord = "hej";
string baklänges = "";

for (int i = ord.Length - 1; i >= 0; i--)
{
    baklänges += ord[i];
}

Console.WriteLine(baklänges); // "jeh"

5. Räkna vokaler i en sträng


string text = "CoderPad";
int count = 0;
string vokaler = "aeiouåäöAEIOUÅÄÖ";

for (int i = 0; i < text.Length; i++)
{
    if (vokaler.Contains(text[i]))
        count++;
}

Console.WriteLine($"Antal vokaler: {count}");

6. Kontrollera om en sträng är ett palindrom


string ord = "level";
bool ärPalindrom = true;

for (int i = 0; i < ord.Length / 2; i++)
{
    if (ord[i] != ord[ord.Length - 1 - i])
    {
        ärPalindrom = false;
        break;
    }
}

Console.WriteLine(ärPalindrom ? "Palindrom" : "Inte palindrom");

7. Balanced Parentheses (med Stack)


using System.Collections.Generic;

string s = "{[()]}";
Stack<char> stack = new Stack<char>();
bool balanced = true;

foreach (char c in s)
{
    if ("({[".Contains(c))
        stack.Push(c);
    else if (")}]".Contains(c))
    {
        if (stack.Count == 0 ||
            (c == ')' && stack.Peek() != '(') ||
            (c == ']' && stack.Peek() != '[') ||
            (c == '}' && stack.Peek() != '{'))
        {
            balanced = false;
            break;
        }
        stack.Pop();
    }
}

Console.WriteLine(balanced && stack.Count == 0
    ? "Balanserad"
    : "Obalanserad");
