7. Balanced Parentheses (med Stack)


using System.Collections.Generic;

string s = "{[()]}";
Stack<char> stack = new Stack<char>();
bool balanced = true;

foreach (char c in s)
{
    if ("({[".Contains(c))
        stack.Push(c);
    else if (")}]".Contains(c))
    {
        if (stack.Count == 0 ||
            (c == ')' && stack.Peek() != '(') ||
            (c == ']' && stack.Peek() != '[') ||
            (c == '}' && stack.Peek() != '{'))
        {
            balanced = false;
            break;
        }
        stack.Pop();
    }
}

Console.WriteLine(balanced && stack.Count == 0
    ? "Balanserad"
    : "Obalanserad");

16. Bubble Sort (sortera array)


int[] tal = { 5, 1, 4, 2, 8 };

for (int i = 0; i < tal.Length - 1; i++)
{
    for (int j = 0; j < tal.Length - i - 1; j++)
    {
        if (tal[j] > tal[j + 1])
        {
            int temp = tal[j];
            tal[j] = tal[j + 1];
            tal[j + 1] = temp;
        }
    }
}

Console.WriteLine(string.Join(", ", tal));

19. LINQ: Filtrera och sortera en lista

20. LINQ: Gruppning och räkning


using System;
using System.Linq;

string text = "hej hej hopp";
var grupper = text.Split(' ')
                  .GroupBy(ord => ord)
                  .Select(g => new { Ord = g.Key, Antal = g.Count() });

foreach (var g in grupper)
{
    Console.WriteLine($"{g.Ord}: {g.Antal}");
}
// hej: 2
// hopp: 1

22. Regex: Hitta alla e-postadresser i en text


using System;
using System.Text.RegularExpressions;

string text = "Kontakta oss på info@test.com eller support@firma.se";
Regex regex = new Regex(@"\b[\w\.-]+@[\w\.-]+\.\w{2,4}\b");

MatchCollection matcher = regex.Matches(text);

foreach (Match m in matcher)
{
    Console.WriteLine(m.Value);
}
// info@test.com
// support@firma.se
